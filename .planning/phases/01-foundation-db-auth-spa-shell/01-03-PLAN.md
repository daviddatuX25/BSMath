Thin fetch wrapper. All requests go to api/ relative path. Always parses JSON envelope.

```js
// assets/js/api.js
// Lightweight fetch wrapper.
// Every response is expected to be { success, data, error }.

const BASE = 'api/';

async function request(method, endpoint, body = null) {
  const options = {
    method,
    credentials: 'same-origin',   // sends session cookie
    headers: { 'Content-Type': 'application/json' },
  };
  if (body !== null) {
    options.body = JSON.stringify(body);
  }
  const res = await fetch(BASE + endpoint, options);
  const json = await res.json();
  // Attach HTTP status so callers can inspect it if needed
  json._status = res.status;
  return json;
}

export const api = {
  get:    (endpoint)       => request('GET',    endpoint),
  post:   (endpoint, body) => request('POST',   endpoint, body),
  put:    (endpoint, body) => request('PUT',    endpoint, body),
  delete: (endpoint)       => request('DELETE', endpoint),
};
```

--- FILE: assets/js/auth.js ---
Manages the current-user cache. Exposes init(), login(), logout(), getUser().

```js
// assets/js/auth.js
// Auth state management.
// _currentUser is the in-memory cache of the logged-in user (or null).

import { api } from './api.js';

let _currentUser = null;

export const auth = {

  // Called once on app boot — tries to restore session from PHP cookie.
  async init() {
    const res = await api.get('auth/me.php');
    _currentUser = res.success ? res.data : null;
    return _currentUser;
  },

  // Returns the cached user (does NOT call server again).
  getUser() {
    return _currentUser;
  },

  // Returns true if there is a logged-in user.
  isLoggedIn() {
    return _currentUser !== null;
  },

  // POST login. On success, caches user and returns it.
  async login(email, password) {
    const res = await api.post('auth/login.php', { email, password });
    if (res.success) {
      _currentUser = res.data;
    }
    return res;
  },

  // POST logout. Clears cache.
  async logout() {
    await api.post('auth/logout.php');
    _currentUser = null;
  },

  // Mounts the login page inside #app and wires the submit handler.
  async mountLoginPage() {
    const app = document.getElementById('app');
    const res  = await fetch('views/login.html');
    app.innerHTML = await res.text();

    const form = document.getElementById('login-form');
    const errEl = document.getElementById('login-error');

    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      errEl.textContent = '';
      const email    = form.email.value.trim();
      const password = form.password.value;
      const result   = await auth.login(email, password);
      if (result.success) {
        // Navigate to dashboard — router will mount shell
        window.location.hash = '#/dashboard';
      } else {
        errEl.textContent = result.error || 'Login failed. Try again.';
      }
    });
  }
};
```

// Track whether the shell is already mounted (avoid re-fetching on every nav)
let _shellMounted = false;
let _shellUser    = null;

async function mountShell(user) {
  // Re-mount if user changed (e.g., after re-login as different role)
  if (_shellMounted && _shellUser?.id === user.id) return;

  const app = document.getElementById('app');
  const res  = await fetch('views/shell.html');
  app.innerHTML = await res.text();

  // Inject user name into topbar
  const nameEl = document.getElementById('topbar-username');
  if (nameEl) nameEl.textContent = user.name;

  // Build role-filtered sidebar nav
  buildSidebar(user.role);

  // Wire logout button
  const logoutBtn = document.getElementById('logout-btn');
  if (logoutBtn) {
    logoutBtn.addEventListener('click', async () => {
      await auth.logout();
      _shellMounted = false;
      _shellUser    = null;
      window.location.hash = '#/login';
    });
  }

  _shellMounted = true;
  _shellUser    = user;
}

const NAV_ITEMS = [
  { hash: '#/dashboard',    label: 'Dashboard',       icon: 'dashboard',        roles: ['admin','dean','program_head'] },
  { hash: '#/programs',     label: 'Manage Programs', icon: 'school',           roles: ['admin','program_head'] },

  const nav = document.getElementById('sidebar-nav');
  if (!nav) return;

  nav.innerHTML = NAV_ITEMS
    .filter(item => item.roles.includes(role))
    .map(item => `
      <a href="${item.hash}"
         data-hash="${item.hash}"
         class="nav-item flex items-center gap-3 px-4 py-2.5 text-stone-300 hover:text-white hover:bg-stone-800 rounded-r-lg mr-2">
        <span class="material-symbols-outlined">${item.icon}</span>
        <span class="text-sm font-medium">${item.label}</span>
      </a>`)
    .join('');

function setActiveNav(hash) {
  document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
  const active = document.querySelector(`.nav-item[data-hash="${hash}"]`);
  if (active) active.classList.add('active');

export const router = {
  navigate(hash) {
    // Normalise: strip query string, default to #/login
    const cleanHash = (hash || '#/login').split('?')[0];
    const route = routes[cleanHash];

    const user = auth.getUser();

    // No matching route → fallback
    if (!route) {
      window.location.hash = user ? '#/dashboard' : '#/login';
      return;
    }

    // Public route (login page)
    if (route.roles.includes('public')) {
      // If already logged in, skip login page
      if (user) {
        window.location.hash = '#/dashboard';
        return;
      }
      route.loader(null);
      return;
    }

    // Protected route — must be authenticated
    if (!user) {
      window.location.hash = '#/login';
      return;
    }

    // Role guard — check if user's role is allowed
    if (!route.roles.includes('*') && !route.roles.includes(user.role)) {
      // Redirect to dashboard (user is authenticated but not allowed here)
      window.location.hash = '#/dashboard';
      return;
    }

    route.loader(user);
  },

  start() {
    // Listen for hash changes
    window.addEventListener('hashchange', () => {
      router.navigate(window.location.hash);
    });
  }
};

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| #hash → router | Client-controlled hash must be validated against route definitions before rendering |
| session cookie → auth.init() | Cookie authenticity relies on PHP session — client cannot forge |
| views/*.html fetch | Static HTML fragments fetched from same origin — no XSS risk from external source |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | router.js role guard | mitigate | Role guard checks both client-side (hide nav) AND server-side PHP checks will be enforced in Phases 2-4 for each API endpoint |
| T-03-02 | Spoofing | auth.js _currentUser cache | mitigate | On every app boot, auth.init() calls me.php to re-validate session — in-memory cache cannot survive refresh |
| T-03-03 | Tampering | views/*.html | accept | Fragments served from same-origin localhost; no user input rendered without escaping in Phase 2+ |
| T-03-04 | Information Disclosure | api.js | accept | API base path 'api/' is not secret; all responses require valid session to return data |
</threat_model>
