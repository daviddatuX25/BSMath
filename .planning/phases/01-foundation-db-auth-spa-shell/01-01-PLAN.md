-- db/schema.sql
-- BS Mathematics Admin SPA — full schema
-- Run this ONCE to create the db_bsmath database and all tables.

CREATE DATABASE IF NOT EXISTS db_bsmath CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE db_bsmath;

DROP TABLE IF EXISTS activities;
DROP TABLE IF EXISTS gallery;
DROP TABLE IF EXISTS faculty;
DROP TABLE IF EXISTS news;
DROP TABLE IF EXISTS events;
DROP TABLE IF EXISTS announcements;
DROP TABLE IF EXISTS programs;
DROP TABLE IF EXISTS users;

CREATE TABLE users (
  id        INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  name      VARCHAR(120) NOT NULL,
  email     VARCHAR(191) NOT NULL UNIQUE,
  password  VARCHAR(255) NOT NULL,          -- bcrypt hash
  role      ENUM('admin','dean','program_head') NOT NULL DEFAULT 'program_head',
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

CREATE TABLE programs (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  title       VARCHAR(191) NOT NULL,
  description TEXT,
  created_by  INT UNSIGNED NOT NULL,
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id)
) ENGINE=InnoDB;

CREATE TABLE announcements (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  title       VARCHAR(191) NOT NULL,
  body        TEXT NOT NULL,
  status      ENUM('pending','approved','rejected') NOT NULL DEFAULT 'pending',
  created_by  INT UNSIGNED NOT NULL,
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id)
) ENGINE=InnoDB;

CREATE TABLE events (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  title       VARCHAR(191) NOT NULL,
  description TEXT,
  event_date  DATE NOT NULL,
  status      ENUM('pending','approved','rejected') NOT NULL DEFAULT 'pending',
  created_by  INT UNSIGNED NOT NULL,
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id)
) ENGINE=InnoDB;

CREATE TABLE news (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  title       VARCHAR(191) NOT NULL,
  body        TEXT NOT NULL,
  created_by  INT UNSIGNED NOT NULL,
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id)
) ENGINE=InnoDB;

CREATE TABLE gallery (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  caption     VARCHAR(255),
  filename    VARCHAR(255) NOT NULL,          -- stored path under uploads/
  created_by  INT UNSIGNED NOT NULL,
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id)
) ENGINE=InnoDB;

CREATE TABLE faculty (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  name        VARCHAR(120) NOT NULL,
  position    VARCHAR(120),
  email       VARCHAR(191),
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;

CREATE TABLE activities (
  id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  user_id     INT UNSIGNED NOT NULL,
  action      VARCHAR(255) NOT NULL,          -- e.g. "Created announcement: Enrollment Open"
  created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
) ENGINE=InnoDB;

  <action>
Create api/connect.php. This file is required (include/require) at the top of every API endpoint. It must:
1. Call session_start() — so every API file gets sessions automatically
2. Connect to db_bsmath via MySQLi
3. Set JSON Content-Type header — so every API file's output is JSON
4. Die with a JSON error response if connection fails

```php
<?php
// api/connect.php
// Require this at the top of every API endpoint.
// Provides: $conn (MySQLi), starts session, sets JSON header.

session_start();

header('Content-Type: application/json');
header('Access-Control-Allow-Origin: *');           // dev only — remove in production
header('Access-Control-Allow-Methods: GET, POST, PUT, DELETE, OPTIONS');
header('Access-Control-Allow-Headers: Content-Type');

// Preflight request — browsers send OPTIONS before cross-origin POST
if ($_SERVER['REQUEST_METHOD'] === 'OPTIONS') {
    http_response_code(200);
    exit;
}

$conn = mysqli_connect('localhost', 'root', '', 'db_bsmath');

if (!$conn) {
    http_response_code(500);
    echo json_encode([
        'success' => false,
        'data'    => null,
        'error'   => 'Database connection failed: ' . mysqli_connect_error()
    ]);
    exit;
}

mysqli_set_charset($conn, 'utf8mb4');
```

Notes:
- session_start() must be called BEFORE any output (header() calls are fine before output).
- The CORS headers are needed because Laragon serves PHP on a port and the browser SPA may be on another port during development.
- Every response from our API uses the envelope: {"success": bool, "data": ..., "error": "..."}
